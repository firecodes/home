# 运行项目

# 克隆整个项目到本地
git clone https://github.com/firecodes/home.git

# 切换到项目所在文件夹
cd home

# 安装依赖
npm install

# 打开浏览器，输入react-native start 会自动
react-native start  
react-native run-android
